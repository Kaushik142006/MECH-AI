import { useState, useRef, useEffect } from "react";
import { ChatMessage as ChatMessageComponent } from "@/components/ChatMessage";
import { TypingIndicator } from "@/components/TypingIndicator";
import { DesignLoader } from "@/components/DesignLoader";
import { ModelViewer } from "@/components/ModelViewer";
import { sendChatMessage, getModelUrl, type ChatMessage } from "@/lib/api";
import { Button } from "@/components/ui/button";
import { Input } from "@/components/ui/input";
import { Send, Cpu } from "lucide-react";

type AppState = "chat" | "designing" | "viewing";

export default function Index() {
  const [state, setState] = useState<AppState>("chat");
  const [messages, setMessages] = useState<ChatMessage[]>([
    { role: "assistant", content: "Hello! I'm **MECHAI**, your AI-powered mechanical design assistant.\n\nWhat are we designing today?" },
  ]);
  const [input, setInput] = useState("");
  const [isTyping, setIsTyping] = useState(false);
  const [stlUrl, setStlUrl] = useState<string | null>(null);
  const scrollRef = useRef<HTMLDivElement>(null);

  useEffect(() => {
    scrollRef.current?.scrollTo({ top: scrollRef.current.scrollHeight, behavior: "smooth" });
  }, [messages, isTyping]);

  const handleSend = async () => {
    const trimmed = input.trim();
    if (!trimmed || isTyping) return;

    const userMsg: ChatMessage = { role: "user", content: trimmed };
    const newHistory = [...messages, userMsg];
    setMessages(newHistory);
    setInput("");
    setIsTyping(true);

    try {
      const response = await sendChatMessage(trimmed, messages);
      const botMsg: ChatMessage = { role: "assistant", content: response.reply };
      setMessages((prev) => [...prev, botMsg]);

      if (response.designing) {
        setState("designing");
        // Poll or wait for STL — the designing flag means the backend is working
        // The stl_url will come in the response when ready
      }

      if (response.stl_url) {
        setStlUrl(getModelUrl(response.stl_url));
        setState("viewing");
      }
    } catch (err) {
      setMessages((prev) => [
        ...prev,
        { role: "assistant", content: "⚠️ Could not connect to the MECHAI backend. Make sure your Python server is running on `localhost:5000`." },
      ]);
    } finally {
      setIsTyping(false);
    }
  };

  const handleKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      handleSend();
    }
  };

  const handleBackToChat = () => {
    setMessages([
      { role: "assistant", content: "Ready for a new design! What would you like to create?" },
    ]);
    setStlUrl(null);
    setState("chat");
  };

  if (state === "designing") return <DesignLoader />;
  if (state === "viewing" && stlUrl) return <ModelViewer stlUrl={stlUrl} onBackToChat={handleBackToChat} />;

  return (
    <div className="flex h-screen flex-col bg-background">
      {/* Header */}
      <header className="flex items-center gap-3 border-b border-border px-6 py-4 glass">
        <div className="flex items-center gap-2">
          <div className="flex h-9 w-9 items-center justify-center rounded-lg bg-primary/15 glow-primary">
            <Cpu className="h-5 w-5 text-primary" />
          </div>
          <div>
            <h1 className="text-lg font-bold tracking-tight text-foreground">
              MEC<span className="text-primary">HAI</span>
            </h1>
            <p className="text-[10px] font-mono uppercase tracking-widest text-muted-foreground">
              Mechanical Design AI
            </p>
          </div>
        </div>
      </header>

      {/* Messages */}
      <div ref={scrollRef} className="flex-1 overflow-y-auto px-4 py-6">
        <div className="mx-auto flex max-w-2xl flex-col gap-4">
          {messages.map((msg, i) => (
            <ChatMessageComponent key={i} role={msg.role} content={msg.content} />
          ))}
          {isTyping && <TypingIndicator />}
        </div>
      </div>

      {/* Input */}
      <div className="border-t border-border px-4 py-4 glass">
        <div className="mx-auto flex max-w-2xl gap-2">
          <Input
            value={input}
            onChange={(e) => setInput(e.target.value)}
            onKeyDown={handleKeyDown}
            placeholder="Describe your part or provide dimensions..."
            className="flex-1 bg-muted/50 border-border/50 focus-visible:ring-primary/50"
            disabled={isTyping}
          />
          <Button onClick={handleSend} disabled={isTyping || !input.trim()} className="glow-primary">
            <Send className="h-4 w-4" />
          </Button>
        </div>
        <p className="mx-auto mt-2 max-w-2xl text-center text-[11px] text-muted-foreground font-mono">
          Connected to local Ollama • build123d pipeline
        </p>
      </div>
    </div>
  );
}

import { cn } from "@/lib/utils";
import ReactMarkdown from "react-markdown";
import { Bot, User } from "lucide-react";

interface ChatMessageProps {
  role: "user" | "assistant";
  content: string;
}

export function ChatMessage({ role, content }: ChatMessageProps) {
  const isUser = role === "user";

  return (
    <div className={cn("flex gap-3 animate-fade-in", isUser ? "flex-row-reverse" : "flex-row")}>
      <div
        className={cn(
          "flex h-8 w-8 shrink-0 items-center justify-center rounded-lg",
          isUser ? "bg-primary/20 text-primary" : "bg-accent/20 text-accent"
        )}
      >
        {isUser ? <User className="h-4 w-4" /> : <Bot className="h-4 w-4" />}
      </div>
      <div
        className={cn(
          "max-w-[75%] rounded-2xl px-4 py-3 text-sm leading-relaxed",
          isUser
            ? "bg-primary text-primary-foreground rounded-tr-sm"
            : "bg-secondary text-secondary-foreground rounded-tl-sm"
        )}
      >
        {isUser ? (
          <p>{content}</p>
        ) : (
          <ReactMarkdown
            components={{
              p: ({ children }) => <p className="mb-2 last:mb-0">{children}</p>,
              strong: ({ children }) => <strong className="font-semibold text-foreground">{children}</strong>,
              ul: ({ children }) => <ul className="list-disc pl-4 mb-2">{children}</ul>,
              li: ({ children }) => <li className="mb-1">{children}</li>,
            }}
          >
            {content}
          </ReactMarkdown>
        )}
      </div>
    </div>
  );
}

import { useEffect, useState } from "react";
import { Cog, Code, Box } from "lucide-react";

const stages = [
  { icon: Cog, label: "Crafting blueprint...", duration: 3000 },
  { icon: Code, label: "Generating code...", duration: 3000 },
  { icon: Box, label: "Building 3D model...", duration: 4000 },
];

export function DesignLoader() {
  const [stageIndex, setStageIndex] = useState(0);

  useEffect(() => {
    if (stageIndex >= stages.length - 1) return;
    const timer = setTimeout(() => setStageIndex((i) => i + 1), stages[stageIndex].duration);
    return () => clearTimeout(timer);
  }, [stageIndex]);

  const current = stages[stageIndex];
  const Icon = current.icon;

  return (
    <div className="fixed inset-0 z-50 flex flex-col items-center justify-center bg-background">
      {/* Glow rings */}
      <div className="relative mb-12">
        <div
          className="absolute inset-0 rounded-full bg-primary/20 blur-3xl"
          style={{ animation: "pulse-glow 2s ease-in-out infinite", width: 200, height: 200, margin: "-60px" }}
        />
        <div
          className="relative flex h-20 w-20 items-center justify-center rounded-full border-2 border-primary/40 bg-card"
          style={{ animation: "spin-slow 8s linear infinite" }}
        >
          <div style={{ animation: "spin-slow 8s linear infinite reverse" }}>
            <Icon className="h-8 w-8 text-primary" />
          </div>
        </div>
      </div>

      {/* Stage label */}
      <p className="text-xl font-medium text-foreground glow-text mb-6 animate-fade-in" key={stageIndex}>
        {current.label}
      </p>

      {/* Progress dots */}
      <div className="flex gap-3">
        {stages.map((_, i) => (
          <div
            key={i}
            className={`h-2 rounded-full transition-all duration-500 ${
              i <= stageIndex ? "w-8 bg-primary" : "w-2 bg-muted"
            }`}
          />
        ))}
      </div>

      <p className="mt-8 text-sm text-muted-foreground font-mono">MECHAI ENGINE</p>
    </div>
  );
}

import { Suspense, useRef, useState } from "react";
import { Canvas } from "@react-three/fiber";
import { OrbitControls, Grid, Environment, Center } from "@react-three/drei";
import { STLLoader } from "three/examples/jsm/loaders/STLLoader.js";
import { useLoader } from "@react-three/fiber";
import * as THREE from "three";
import { Button } from "@/components/ui/button";
import { RotateCcw, Grid3x3, Download, MessageSquare } from "lucide-react";

function STLModel({ url, wireframe }: { url: string; wireframe: boolean }) {
  const geometry = useLoader(STLLoader, url);
  const meshRef = useRef<THREE.Mesh>(null);

  // Center and scale
  geometry.computeBoundingBox();
  const box = geometry.boundingBox!;
  const size = new THREE.Vector3();
  box.getSize(size);
  const maxDim = Math.max(size.x, size.y, size.z);
  const scale = 3 / maxDim;

  return (
    <Center>
      <mesh ref={meshRef} geometry={geometry} scale={scale} castShadow receiveShadow>
        <meshStandardMaterial
          color={wireframe ? "#00cc88" : "#22d3a0"}
          wireframe={wireframe}
          metalness={0.3}
          roughness={0.4}
        />
      </mesh>
    </Center>
  );
}

function LoadingFallback() {
  return (
    <mesh>
      <boxGeometry args={[1, 1, 1]} />
      <meshStandardMaterial color="#22d3a0" wireframe />
    </mesh>
  );
}

interface ModelViewerProps {
  stlUrl: string;
  onBackToChat: () => void;
}

export function ModelViewer({ stlUrl, onBackToChat }: ModelViewerProps) {
  const [wireframe, setWireframe] = useState(false);
  const controlsRef = useRef<any>(null);

  const handleReset = () => {
    if (controlsRef.current) {
      controlsRef.current.reset();
    }
  };

  const handleDownload = () => {
    const a = document.createElement("a");
    a.href = stlUrl;
    a.download = "mechai-model.stl";
    a.click();
  };

  return (
    <div className="flex h-screen flex-col bg-background">
      {/* Toolbar */}
      <div className="flex items-center justify-between border-b border-border px-4 py-3 glass">
        <div className="flex items-center gap-2">
          <div className="h-3 w-3 rounded-full bg-primary animate-pulse" />
          <span className="font-mono text-sm text-primary">MECHAI</span>
          <span className="text-xs text-muted-foreground">/ Model Viewer</span>
        </div>
        <div className="flex items-center gap-2">
          <Button variant="ghost" size="sm" onClick={handleReset}>
            <RotateCcw className="h-4 w-4" />
            Reset
          </Button>
          <Button
            variant={wireframe ? "secondary" : "ghost"}
            size="sm"
            onClick={() => setWireframe(!wireframe)}
          >
            <Grid3x3 className="h-4 w-4" />
            Wireframe
          </Button>
          <Button variant="ghost" size="sm" onClick={handleDownload}>
            <Download className="h-4 w-4" />
            Download
          </Button>
          <Button variant="outline" size="sm" onClick={onBackToChat} className="border-primary/30 text-primary hover:bg-primary/10">
            <MessageSquare className="h-4 w-4" />
            New Design
          </Button>
        </div>
      </div>

      {/* 3D Canvas */}
      <div className="flex-1">
        <Canvas
          shadows
          camera={{ position: [5, 5, 5], fov: 45 }}
          style={{ background: "hsl(220, 20%, 7%)" }}
        >
          <ambientLight intensity={0.4} />
          <directionalLight position={[10, 10, 5]} intensity={1} castShadow />
          <directionalLight position={[-5, 5, -5]} intensity={0.3} />
          <pointLight position={[0, 10, 0]} intensity={0.5} color="#22d3a0" />

          <Suspense fallback={<LoadingFallback />}>
            <STLModel url={stlUrl} wireframe={wireframe} />
          </Suspense>

          <Grid
            infiniteGrid
            cellSize={0.5}
            cellThickness={0.5}
            sectionSize={2}
            sectionThickness={1}
            cellColor="#1a2a3a"
            sectionColor="#243040"
            fadeDistance={20}
            position={[0, -2, 0]}
          />

          <OrbitControls
            ref={controlsRef}
            enableDamping
            dampingFactor={0.05}
            minDistance={2}
            maxDistance={20}
          />
        </Canvas>
      </div>
    </div>
  );
}

const API_BASE = "http://localhost:5000";

export interface ChatMessage {
  role: "user" | "assistant";
  content: string;
}

export interface ChatResponse {
  reply: string;
  stl_url?: string;
  designing?: boolean;
}

export async function sendChatMessage(
  message: string,
  history: ChatMessage[]
): Promise<ChatResponse> {
  const res = await fetch(`${API_BASE}/api/chat`, {
    method: "POST",
    headers: { "Content-Type": "application/json" },
    body: JSON.stringify({ message, history }),
  });

  if (!res.ok) {
    throw new Error(`API error: ${res.status}`);
  }

  return res.json();
}

export function getModelUrl(filename: string): string {
  return `${API_BASE}/api/model/${filename}`;
}

"""
MECHAI Backend Server
=====================
Run this on your local machine where Ollama and build123d are installed.

Setup:
  pip install flask flask-cors openai build123d

Usage:
  1. Start Ollama:  ollama serve
  2. Pull models:   ollama pull llama3.1 && ollama pull llama3.2
  3. Run server:    python mechai_server.py

The server runs at http://localhost:5000
"""

from flask import Flask, request, jsonify, send_file
from flask_cors import CORS
from openai import OpenAI
import os, re, traceback

app = Flask(__name__)
CORS(app)  # Allow all origins (Lovable preview)

client = OpenAI(base_url="http://localhost:11434/v1", api_key="ollama")

MODEL_DIR = os.path.join(os.path.dirname(__file__), "models")
os.makedirs(MODEL_DIR, exist_ok=True)


# ─── Agent 1: Intro Bot (Parameter Collector) ───────────────────────────

INTRO_SYSTEM = """You are the MECHAI Parameter Collector. Your sole purpose is to gather geometric dimensions for 3D modeling.

CRITICAL BEHAVIOR:
1. MINIMALISM: If the user provides dimensions (e.g., "10mm inner, 30mm outer, 20mm thick"), DO NOT ask about material, safety factors, or usage.
2. THE "SHAPE" RULE: After dimensions are provided, you may ask ONLY ONE clarifying question regarding the profile (e.g., "Hexagonal or Circular?").
3. NO REPETITION: Once a value is given, never ask for it again.
4. TRIGGER: As soon as you have the dimensions and the shape, stop asking questions immediately.

REQUIRED PARAMETERS BY CATEGORY:
- Fasteners (Nuts/Bolts): Inner Dia, Outer Dia, Thickness/Length, Shape.
- Basic Shapes (Boxes/Cylinders): L, W, H or Dia, Height.
- Complex (Helmets/Custom): Overall Scale (Small/Medium/Large).

RESPONSE PROTOCOL:
- Greeting: Short ("Hello! What are we designing today?")
- Acknowledgment: "Ok, noted."
- Completion: Once all dimensions + shape are known, respond EXACTLY with:
  "All required parameters collected.
  Summary:
  - Object: [Name]
  - Dimensions: [List values]
  - Shape: [Value]
  Now the design process will be started."

STRICT PROHIBITIONS:
- NEVER ask about Material Type.
- NEVER ask about Safety Factors or Stress.
- NEVER explain how CAD works.
- NEVER generate code."""


# ─── Agent 2: Prompt Architect ──────────────────────────────────────────

def prompt_agent(summary):
    system = """You are the MECHAI Prompt Architect.
Your task: Convert a user's dimension summary into a precise technical prompt for a CAD Coder.

Rules for the Technical Prompt:
1. Specify the library: 'build123d'.
2. Specify the method: 'Builder Mode' (using with BuildPart() as result:).
3. List exact variables: Define dimensions as float variables first.
4. Structural Requirement: 'First create the base shape, then extrude, then subtract the center hole.'
5. Constraint: 'Use ONLY Capitalized classes like Extrude, Hole, and RegularPolygon.'

Output ONLY the final prompt. No conversation."""

    res = client.chat.completions.create(
        model="llama3.1",
        messages=[
            {"role": "system", "content": system},
            {"role": "user", "content": f"Create a CAD prompt for this summary: {summary}"},
        ],
    )
    return res.choices[0].message.content


# ─── Agent 3: Coder ─────────────────────────────────────────────────────

def coder_agent(blueprint):
    system = """You are the MECHAI CAD Coder.
Task: Write a single Python script using 'build123d' based on the technical blueprint.

STRICT RULES:
1. Start with: 'from build123d import *'
2. Constraint: 'Use ONLY Capitalized classes like Extrude, Hole, and RegularPolygon.'
3. Final object MUST be named 'result'.
4. End with: 'result.part.export_stl("OUTPUT_PATH")'
5. Output ONLY the raw Python code. No markdown backticks, no comments, no explanation."""

    res = client.chat.completions.create(
        model="llama3.1",
        messages=[
            {"role": "system", "content": system},
            {"role": "user", "content": f"Generate the code for this blueprint: {blueprint}"},
        ],
        temperature=0,
    )
    return res.choices[0].message.content.strip()


# ─── Execution Engine ───────────────────────────────────────────────────

def execute_design(code):
    from build123d import (
        BuildPart, BuildSketch, RegularPolygon, Circle,
        extrude, Hole, Axis, Plane
    )

    stl_path = os.path.join(MODEL_DIR, "model.stl")

    # Clean markdown fences
    clean = re.sub(r"```python|```", "", code).strip()
    # Inject correct output path
    clean = clean.replace('export_stl("model.stl")', f'export_stl("{stl_path}")')

    if os.path.exists(stl_path):
        os.remove(stl_path)

    local_scope = {
        "BuildPart": BuildPart, "BuildSketch": BuildSketch,
        "RegularPolygon": RegularPolygon, "Circle": Circle,
        "Extrude": extrude, "Hole": Hole, "Axis": Axis, "Plane": Plane,
    }

    exec(clean, globals(), local_scope)

    if os.path.exists(stl_path):
        return "model.stl"
    return None


# ─── API Routes ─────────────────────────────────────────────────────────

@app.route("/api/chat", methods=["POST"])
def chat():
    data = request.json
    user_message = data.get("message", "")
    history = data.get("history", [])

    # Build messages for Ollama
    messages = [{"role": "system", "content": INTRO_SYSTEM}]
    for h in history:
        messages.append({"role": h["role"], "content": h["content"]})
    messages.append({"role": "user", "content": user_message})

    # Call Intro Bot
    res = client.chat.completions.create(model="llama3.2", messages=messages)
    bot_reply = res.choices[0].message.content

    # Check if parameters are collected → run the full pipeline
    if "All required parameters collected" in bot_reply:
        try:
            print("[MECHAI] Parameters collected. Running pipeline...")
            blueprint = prompt_agent(bot_reply)
            print(f"[MECHAI] Blueprint:\n{blueprint[:200]}...")
            code = coder_agent(blueprint)
            print(f"[MECHAI] Code:\n{code[:200]}...")
            stl_file = execute_design(code)

            if stl_file:
                bot_reply += "\n\n✅ **Success!** Your 3D model has been generated."
                return jsonify({"reply": bot_reply, "stl_url": stl_file})
            else:
                bot_reply += "\n\n⚠️ Model generation failed. Please try again with different dimensions."
        except Exception as e:
            print(f"[MECHAI] Pipeline error: {traceback.format_exc()}")
            bot_reply += f"\n\n⚠️ Error during generation: {str(e)}"

    return jsonify({"reply": bot_reply})


@app.route("/api/model/<filename>", methods=["GET"])
def serve_model(filename):
    filepath = os.path.join(MODEL_DIR, filename)
    if os.path.exists(filepath):
        return send_file(filepath, mimetype="application/octet-stream")
    return jsonify({"error": "File not found"}), 404


if __name__ == "__main__":
    print("=" * 50)
    print("  MECHAI Backend Server")
    print("  http://localhost:5000")
    print("=" * 50)
    app.run(host="0.0.0.0", port=5000, debug=True)

HOW IT WORKS:
  1. User chats in the React UI (Lovable preview)
  2. Messages are sent to your Python server (localhost:5000)
  3. Agent 1 (Intro Bot) collects dimensions via Ollama

     → Agent 3 (Coder) generates build123d Python code
     → Execution engine runs the code and exports STL
  5. STL file is sent back to the React frontend
  6. Three.js renders the 3D model with orbit controls
